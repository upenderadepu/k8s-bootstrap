VERIFY INSTALLATION

  # Controller pod running in tenant control plane:
  kubectl -n {{ .Release.Namespace }} get pods -l app=kubevirt-csi-driver

  # Node plugin DaemonSet healthy on all workers:
  kubectl -n {{ .Release.Namespace }} get daemonset kubevirt-csi-node

  # CSIDriver registered:
  kubectl get csidriver csi.kubevirt.io

  # StorageClass(es) present:
  kubectl get storageclass

  # Smoke-test — create a PVC and verify it binds:
  kubectl apply -f - <<'EOF'
  apiVersion: v1
  kind: PersistentVolumeClaim
  metadata:
    name: csi-test-pvc
  spec:
    accessModes: [ReadWriteOnce]
    resources:
      requests:
        storage: 1Gi
    storageClassName: {{ (index .Values.storageClasses 0).name | default "kubevirt" }}
  EOF
  kubectl wait --for=condition=Bound pvc/csi-test-pvc --timeout=120s
